243 function getfilter()
244 {
245  $datebegin = GETPOST("datebeginyear") . '-' . GETPOST("datebeginmonth") . '-' . GETPOST("datebeginday");
246  if ($datebegin == '--') { // not in Form check URL
247  $datebegin = GETPOST("datebegin");
248  }
249  if (empty($datebegin)) {
250  $datebegin = date("Y-m-d", strtotime("now - 1 month"));
251  }
252 
253  $dateend = GETPOST("dateendyear") . '-' . GETPOST("dateendmonth") . '-' . GETPOST("dateendday");
254  if ($dateend == '--') { // not in Form check URL
255  $dateend = GETPOST("dateend");
256  }
257  if (empty($dateend)) {
258  $dateend = date("Y-m-d");
259  }
260 
261  $products = GETPOST('multiproducts', 'array');
262  if (empty($products)) { // not in Form check URL
263  $products = GETPOST('products', 'alpha');
264  if (empty($products)) {
265  $selectedproducts = "product.ref IS NOT NULL";
266  } else {
267  $selectedproducts = " product.ref IN " . $products;
268  $products = explode("','", trim($products,"('')"));
269  }
270  } else {
271  $selectedproducts = " product.ref IN ('" . implode("','", $products) . "')";
272  }
273 
274  $filter = array(
275  "datebegin" => $datebegin,
276  "dateend" => $dateend,
277  "products" => $products,
278  "orders" => array(
279  " commande.date_commande >= '" . $datebegin . "' ",
280  " commande.date_commande <= '" . $dateend . "' ",
281  $selectedproducts
282  ),
283  "shipments" => array(
284  "((shipment.date_expedition IS NULL AND shipment.date_creation >= '" . $datebegin . "') OR (shipment.date_expedition >= '" . $datebegin . "'))",
285  "((shipment.date_expedition IS NULL AND shipment.date_creation <= '" . $dateend . "') OR (shipment.date_expedition <= '" . $dateend . "'))",
286  $selectedproducts
287  )
288  );
289 
290  return $filter;
291 }
